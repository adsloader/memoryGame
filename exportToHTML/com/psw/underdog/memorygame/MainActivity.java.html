30       public void addListenerOnButton() { 
31    
32           //Select a specific button to bundle it with the action you want 
33           button = (Button) findViewById(R.id.btnMemory); 
34    
35           button.setOnClickListener(new View.OnClickListener() { 
36    
37               @Override 
38               public void onClick(View view) { 
39                   showMemoryNumber(); 
40               } 
41    
42           }); 
43    
44       }
 
45    
46       private void showMemoryNumber() { 
47           // 기존 메모리 정보를 지운다 
48           mNumberList.clear(); 
49    
50           // MAX_COUNT만큼 저장한다. 
51           for(int i = 0 ; i < MAX_COUNT; i++){ 
52               int nNumber = giveMeANumber(); 
53               mNumberList.add(nNumber); 
54           } 
55    
56           playAnimation(); 
57    
58       }
 
59    
60       /** 
61        *  지금은 이해하기 힘들 것임. 
62        *  Thread와 Timer내에서 UI를 처리하는 
63        *  교육용 예제임. 
64        * */ 
65       private void playAnimation() { 
66    
67           Timer myTimer = new Timer(); 
68           myTimer.schedule(new TimerTask() { 
69               int nIndx = 0; 
70    
71               @Override 
72               public void run() { 
73                   if (nIndx > mNumberList.size() - 1) { 
74                       cancel(); 
75                       return; 
76                   } 
77    
78                   // Android를 경악하게 만드는 것 중에 하나.. 
79                   // Timer나 Thread 내에서 UI를 액세스할 경우, 이런 식의 처리를 해주어야 한다. 
80                   runOnUiThread(new Runnable() { 
81                       @Override 
82                       public void run() { 
83                           if (nIndx > mNumberList.size() - 1) return; 
84    
85                           // TextView에 문자를 찍는다. 
86                           displayNumberToText(nIndx); 
87                           nIndx++; 
88                       } 
89                   }); 
90    
91    
92               } 
93           }, 0, 1000);  // 1초 단위로... 
94    
95       }
 
96    
97       /** 
98        * random 함수로 값을 가져온다. 
99        *  (*)값 이후 상수가 0~부터 값-1까지 이다. 
100       * */ 
101      private int giveMeANumber() { 
102          return (int) (Math.random() * 10); 
103      }
 
104   
105      /** 
106       * 화면에 번호를 출력한다.. 
107       * */ 
108      public void displayNumberToText(int nNumber){ 
109          int n = mNumberList.get(nNumber); 
110          TextView txtView = (TextView)findViewById(R.id.txtNumber); 
111          txtView.setText(">" + n + "<"); 
112      }
 
113   
114      public void OnPress(View v){ 
115          Button btn = (Button)v; 
116          String str = btn.getText().toString(); 
117          int nNumber = Integer.valueOf(str); 
118   
119          CheckMemoryNumber(nNumber); 
120      }
 
121   
122      private void CheckMemoryNumber(int n) { 
123   
124          // 더이상 값이 없다면... 
125          int nSize = mNumberList.size(); 
126          if (nSize < 1){ 
127              // Complete 
128              ShowMessage("완료했습니다. memory를 눌러 다시 시작해주세요..."); 
129              return; 
130          } 
131   
132          int nValue = mNumberList.get(0); 
133          if(nValue == n){ 
134   
135              if(nSize > 1){ 
136                  ShowMessage(n + " 값이 맞습니다. 다음을 맞추어주세요"); 
137              } else { 
138                  ShowMessage("완료했습니다. memory를 눌러 다시 시작해주세요..."); 
139              } 
140              mNumberList.remove(0); 
141          } 
142      }
 
143   
144      // Message를 출력한다. 
145      private void ShowMessage(String str){ 
146          Toast.makeText(getApplicationContext(), str, Toast.LENGTH_LONG).show(); 
147      }
